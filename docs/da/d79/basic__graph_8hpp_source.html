149template <typename NodeInfo>
150basic_graph<NodeInfo> read_basic_graph(std::istream &is) {
151 basic_graph<NodeInfo> result{};
152 bool reading_nodes = true;
153 for (std::string s; std::getline(is, s);) {
154 if (s.empty()) {
155 reading_nodes = false;
156 continue;
157 }
158
159 std::istringstream iss{s};
160 if (reading_nodes) {
161 math::vector2d p;
162 NodeInfo ni{};
163 if (iss >> p && iss >> ni) {
164 result.add_node(p, ni);
165 } else {
166 throw std::runtime_error{"Error while reading node: " + s};
167 }
168 } else {
169 int from;
170 int to;
171 float weight;
172 if (iss >> from && iss >> to && iss >> weight) {
173 auto &from_node = result[from];
174 from_node.add_edge(result[to], weight);
175 } else {
176 throw std::runtime_error{"Error while reading edge: " + s};
177 }
178 }
179 }
180
181 return result;
182}

184template <typename NodeInfo>
185std::pair<float, float> weight_range(basic_graph<NodeInfo> const &graph) {
186 auto[min, max] = std::minmax_element(graph.begin(), graph.end(),
187 [](auto const &l, auto const &r) {
188 return l[0].weight() < r[0].weight();
189 });
190 return std::make_pair(min->operator[](0).weight(),
191 max->operator[](0).weight());
